CREATE TABLE Students(
    id INT,
    name VARCHAR(30),
    birth_date DATE ,
    phone INT,
    GENDER VARCHAR(1)
    
);
CREATE TABLE products(
    id int NOT NULL UNIQUE AUTO_INCREMENT,
    product_name VARCHAR(30),
    product_details TEXT,
    product_price INT,
    PRIMARY KEY (id)
    
    
);
